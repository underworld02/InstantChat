from selenium import webdriver
from selenium.webdriver.edge.options import Options
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import time
from openpyxl import load_workbook

# 配置 Edge 浏览器选项以连接到已打开的浏览器
options = Options()
options.use_chromium = True
options.add_experimental_option("debuggerAddress", "127.0.0.1:9222")

# 创建 WebDriver 实例并连接到已经打开的浏览器
driver = webdriver.Edge(options=options)

def log_info(message):
    """记录信息日志的函数"""
    print(f"Info: {message}")

def log_error(message):
    """记录错误日志的函数"""
    print(f"Error: {message}")

def retry_find_element(driver, by, value, retries=5, wait_time=2):
    """尝试重新定位元素的函数"""
    for i in range(retries):
        try:
            element = WebDriverWait(driver, 10).until(EC.presence_of_element_located((by, value)))  # 等待元素出现
            return element
        except Exception as e:
            if i < retries - 1:  # 如果不是最后一次尝试，就等待并重新定位
                log_info(f"Retrying ({i + 1}/{retries}) to find element: {value}")
                time.sleep(wait_time)  # 延迟 wait_time 秒后再试
            else:
                log_error(f"Element not found after {retries} retries: {value}")
                raise e  # 最后一次尝试失败后抛出异常

def execute_step(step_func, step_name, retries=3, wait_time=2):
    """执行步骤并处理异常的函数"""
    for attempt in range(retries):
        try:
            log_info(f"开始执行步骤: {step_name}")
            step_func()
            log_info(f"步骤 '{step_name}' 成功完成。")
            return
        except Exception as e:
            log_error(f"步骤 '{step_name}' 失败: {e}")
            if attempt < retries - 1:
                log_info(f"等待 {wait_time} 秒后重试 (第 {attempt + 1} 次尝试)...")
                time.sleep(wait_time)
            else:
                log_error(f"步骤 '{step_name}' 多次尝试失败，停止执行。")
                raise

def click_save_draft_button():
    """点击保存草稿按钮"""
    save_draft_button = retry_find_element(driver, By.XPATH, "//button[@data-tracking-id='RfK5PxT202f3Hi0U' and @data-testid='beast-core-button']//span[text()='保存草稿']")
    save_draft_button.click()
    log_info("Clicked the '保存草稿' button successfully.")

def click_confirm_button():
    """点击确认按钮"""
    confirm_button = retry_find_element(driver, By.XPATH, "//button[@data-tracking-id='XNapV2hfCR6zpkdz' and @data-testid='beast-core-button']//span[text()='确认']")
    confirm_button.click()
    log_info("Clicked the '确认' button successfully.")

def delete_second_row_in_excel(file_path):
    """删除Excel文件中的第二行并保存"""
    workbook = load_workbook(file_path)
    sheet = workbook.active
    sheet.delete_rows(2)
    workbook.save(file_path)
    log_info(f"Second row deleted and file saved successfully: {file_path}")

# 使用 execute_step 执行各个步骤
execute_step(click_save_draft_button, "点击保存草稿按钮")
execute_step(click_confirm_button, "点击确认按钮")
